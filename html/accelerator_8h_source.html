00083 inline accelerator_t *ring_get_accelerator(int cpu, int type) {
00084     if(type < 0 || type >= ACC_MAX) return NULL;
00085 
00086     // Find the start point to search for an accelerator
00087     unsigned int start = 0;
00088     for(;start<_ring_layout_size;start++) {
00089         if(_ring_layout[start].core == cpu) break;

00091 
00092     // Now do a wrap around search for the first accelerator that is available and of the right type
00093     for(unsigned int i=0;i<_ring_layout_size;i++) {
00094         if(_ring_layout[(start+i)%_ring_layout_size].type == type)
00095             return &_ring_layout[(start+i)%_ring_layout_size];
