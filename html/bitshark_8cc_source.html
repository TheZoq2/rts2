00671 /**
00672  * Internal utility function: set up the internal CORDIC to shift the spectrum by the requested offset.
00673  * This might attenuate the signal by a bit but functions as a work-around to the DC gap.
00674  * @param bs Handle to the Bitshark device control registers
00675  * @param offset Shift/rotate the spectrum by this much, in Hz
00676  * @param adclk Samplerate from the ADC on the Bitshark (before decimation)
00677  * @param verbose Debug the inner workings by printing internal steps and values
00678  * @return Actual offset implemented by the hardware
00679  */
00680 float bitshark_shift_spectrum(bitshark_dev_t *bs, float offset, unsigned adclk, int verbose) {
00681     // Obtain the step size for the phasor to approximate the desired offset
00682     // Note: invert as the complex carrier should be the opposite of the desired offset
00683     float stepsize = -offset / (float)adclk;
00684     // Convert to fixed point
00685     int phasor = (int)(stepsize * (float)(1 << 23));
00686     // The fixed point is an approximation, calculate the actual offset generated this way
00687     float real_off = ((float)phasor / (float)(1 << 23)) * (float)adclk;
00688     // Invert as the carrier was the opposite of the offset
00689     real_off = -real_off;
00690 
00691     // Set the phasor step
00692     bs->ctrl->retune_reg = phasor;
00693     // Enable the shift by enabling the CORDIC
00694     bitshark_set_retune(bs, 1);
00695 
00696     if(verbose) printf("bitshark_shift_spectrum: target offset: %.0f kHz - step size: %f (@ %u Hz ADC) - actual offset: %.3f kHz\n", offset/1e3f, stepsize, adclk ,real_off/1e3f);
00697     return real_off;
00698 }

00700 /**
00701  * Utility function to quickly tune into a specific frequency with appropriate bandwidth selection. The filtering of
00702  * the signal is done with a mix of analog and digital filters.
00703  * As the MicroBlaze processor cannot handle capture speeds in excess of 5 MS/s (which means 20 cycles per sample), the
00704  * sample rate of the A/D converter will be set to ~5 MS/s which will be decimated to approximately the desired sample rate.
00705  *
00706  * The frequency range is 300 MHz to 4000 MHz in kHz steps, the bandwidth selection ranges from 1 kHz to 2.5 MHz.
00707  * Note: that the bandwidth selection has to be within the bandwidth dictated by the sample rate and the Nyquist frequency.
00708  * Note: the digital filters are designed for sample rates up to 10 MS/s, as such, the A/D speed is fixed on ~5 MS/s.
00709  * @param bs Handle to the Bitshark device
00710  * @param rate_Ss Sample rate per second, range: 19,532 to 5,000,000 - the value will be updated with the actual sample rate
00711  * @param freq_kHz The frequency in kHz ranging from 300 MHz to 2 GHz
00712  * @param bw_kHz The bandwidth, the actual bandwidth will be at most Fsample/2 to prevent aliasing
00713  * @param real_rate_Ss Pointer to a unsigned int which will contain the actual sample rate (as the sample rate will be approximated)
00714  * @param verbose Make the tuning verbose for debugging purposes
00715  * @return 0 if all ok or an error code otherwise
00716  */
00717 int bitshark_auto_tune(bitshark_dev_t *bs, unsigned int rate_Ss, unsigned int freq_kHz, unsigned int bw_kHz, unsigned int *real_rate_Ss, int verbose) {
00718     // Sanity checking: determine if the sample rate is sane
00719     if(rate_Ss < 19532 || rate_Ss > 105000000) {
00720         printf("bitshark_auto_tune error: Requested sample rate is out of range [19532, 105000000]: %u S/s\n", rate_Ss);
00721         return BS_ERR_SAMPLE_RATE;
00722     }
00723 
00724     if(freq_kHz < 300000 || freq_kHz > 4000000) {
00725         printf("bitshark_auto_tune error: Requested frequency is out of range [300000 kHz, 4000000 kHz]: %u kHz\n", freq_kHz);
00726         return BS_ERR_FREQ;
00727     }
00728 
00729     unsigned int Fsample_kHz = rate_Ss / 1000;
00730     if(bw_kHz < 1 || bw_kHz > Fsample_kHz / 2) {
00731         printf("bitshark_auto_tune error: Requested bandwidth is out of range [1 kHz, %u kHz @ %u S/s]: %u kHz\n", Fsample_kHz/2, rate_Ss, bw_kHz);
00732         return BS_ERR_BW;
00733     }
00734 
00735     // Disable retuning as used by auto_tune2
00736     bitshark_set_retune(bs, 0);
00737 
00738     // Tune into the correct frequency
00739     bitshark_cmd_write_freq(bs, freq_kHz, verbose);
00740 
00741     // If we have a sample rate over 5 MS/s we use the Bitshark itself and bypass the logic in the FPGA
00742     if(rate_Ss >= 5000000) {
00743         // Disable/bypass the FIR filter
00744         bitshark_set_lpf(bs, 0);
00745         if(verbose) printf("bitshark_auto_tune: Disabling FIR filter\n");
00746 
00747         // Disable decimation
00748         bitshark_set_decimate(bs, 0);
00749         if(verbose) printf("bitshark_auto_tune: Disabling decimate\n");
00750 
00751         // Set the frequency
00752         bitshark_cmd_write_adclk(bs, rate_Ss, verbose);             // >= 5 MS/s
00753         *real_rate_Ss = rate_Ss;
00754 
00755         // Sanitize the bandwidth to 660 kHz, 1320 kHz, 2000 kHz to 56000 kHz in 1000kHz steps
00756         printf("bitshark_auto_tune: Input bandwidth: %i kHz\n", bw_kHz);
00757         bw_kHz = bitshark_validate_bandwidth(bw_kHz*2);
00758         printf("bitshark_auto_tune: Sanitized bandwidth: %i kHz\n", bw_kHz/2);
00759 
00760         // Set the bandwidth
00761         if(bitshark_cmd_write_chanbw(bs, bw_kHz, verbose) == 0) return BS_ERR_OTHER;
00762     } else {
00763         if(verbose) printf("bitshark_auto_tune: Using downsampling!\n");
00764         // To provide a signal at lower sample rates (lower than 5 MS/s), we will use the FIR filter in a low-pass configuration and
00765         // use integer decimation to obtain lower sample rates.
00766 
00767         // Determine which decimation factor we need and what the base sample frequency will become.
00768         unsigned int dec = (5000000 / rate_Ss);
00769         // If this is the lower boundary (indicated by a remainder), lower the decimation one step. This will
00770         // allow us to use a higher sample frequency on the Bitshark itself.
00771         if(5000000 % rate_Ss > 0) dec++;
00772 
00773         // With this new decimation factor, calculate the required capture frequency in kHz.
00774         unsigned int bs_adclk_kSs = (dec * rate_Ss) / 1000;
00775         // Update the rate to the actual value we deliver
00776         unsigned int samplerate = (bs_adclk_kSs * 1000) / dec;
00777 
00778         // Over/underflow protection
00779         if(dec > 256) dec = 256;
00780         // Set the real sample rate based on the decimation factor
00781         if(real_rate_Ss != NULL) *real_rate_Ss = samplerate;
00782         if(verbose) printf("bitshark_auto_tune: Requested samplerate: %i Hz - Actual: %i Hz (%i kHz AD, %i decimation)\n", rate_Ss, samplerate, bs_adclk_kSs, dec);
00783 
00784         // Safety check: the FIR filters can only handle rates up to 10 MS/s
00785         if(bs_adclk_kSs > 10000) {
00786             printf("bitshark_auto_tune error: the requested integer downsampled frequency requires a sample rate which is too high for the FIR filters: %i kS/s - max: 10000 kS/s", bs_adclk_kSs);
00787             return BS_ERR_OTHER;
00788         }
00789 
00790         // Calculate the optimum bandwidth (assuming we want Wn=0.5) for the Bitshark module based on the AD clock
00791         // Note: do not replace bw_kHz as the FIR filter - below - can more precisely handle the low pass stuff.
00792         unsigned bw = bitshark_validate_bandwidth(bs_adclk_kSs);
00793         if(bw > bs_adclk_kSs) {
00794             printf("bitshark_auto_tune bug: the generated bandwidth setting violates Niquist!\n");
00795             abort();
00796         }
00797 
00798         // Set the A/D and Bitshark filter to the calculated values and set the filter to something sane as well
00799         bitshark_cmd_write_adclk(bs, bs_adclk_kSs * 1000, verbose); // Somewhere above 5 MS/s
00800         bitshark_cmd_write_chanbw(bs, bw, verbose);
00801 
00802         // Set the decimation factor
00803         bitshark_set_decimate(bs, (dec-1));
00804 
00805         // Enable the FIR filter flow
00806         bitshark_set_lpf(bs, 1);
00807 
00808         // Calculate 64 FIR filter coefficients
00809         float coef[64];
00810         float range = (float)bw_kHz / (float)(bs_adclk_kSs * 0.5f); // 0.0 to 1.0 which matches 0 Hz to Fs/2
00811 
00812         printf("Range for FIR filter: ADCLK: %u kHz, BW: %u kHz, %f\n", bs_adclk_kSs, bw_kHz, range);
00813         if(fir_lpf_gen_sinc(64, range, coef) != 0) {
00814             printf("bitshark_auto_tune error: Calculating filter coefficients failed\n");
00815             return BS_ERR_OTHER;
00816         }
00817 
00818         // Load all coefficients
00819         if(fir_load_coef(bs, 64, coef, verbose) != 0) {
00820             printf("Error loading coefficients into FIR filter\n");
00821             return BS_ERR_OTHER;
00822         }
00823     }
00824 
00825     // Enable channel balancing by default; this might attenuate by -3dB
00826     iq_t iqbal;
00827     bitshark_cmd_get_iqbal(bs, freq_kHz, &iqbal, verbose);  // Get channel correction settings from the Bitshark
00828     bitshark_load_iqbal(bs, &iqbal, verbose);                   // Apply them to the balancer
00829     bitshark_set_iqbal(bs, 1);                                  // Enable channel balancing
00830 
00831     // Clear the fifo buffer
00832     bitshark_clear(bs);
00833 
00834     return BS_NO_ERR;
00835 }

00837 /**
00838  * Utility function to quickly tune into a specific frequency with appropriate bandwidth selection. The filtering of
00839  * the signal is done with a mix of analog and digital filters.
00840  *
00841  * This function handles the automatic spectrum shift to work around the DC gap of the physical receiver. Note that
00842  * this means the derived A/D clock has to be supported by the FIR filters: you will get an error if this is not the case.
00843  *
00844  * The actual tuning is done by assuming that the band between -Fs/4 and Fs/4 is actually desired. As the DC gap will ruin the
00845  * data, the tuner is set to roughly (Ftune - Fs/4) and the sample speed is increased. Using the internal CORDIC, the complex
00846  * spectrum is shifted by -Fs/4 and filtered down to the band from -Fs/4 to Fs/4 using the FIR filters and decimated to the
00847  * requested samplerate. As such, this whole chain is invisible to the programmer when this function is used.
00848  *
00849  * WARNING: Because we attempt to approximate the sample frequency and are restricted by the integer decimation, its near
00850  * impossible to skip the DC gap entirely: it *will* show up in the spectrum at the negative side of the bandwidth. Make
00851  * sure you are aware of the 30kHz band with -40dB atenuation at the edge.
00852  *
00853  * The settings used for the various parts of the receiver will result in an approximation of the requested parameters;
00854  * use the feedback variables to make sure the result is what you expect.
00855  *
00856  * Note: that the bandwidth selection has to be within the bandwidth dictated by the sample rate and the Nyquist frequency.
00857  *
00858  * Note: the digital filters are designed for sample rates up to 24 MS/s which is a limit in the tuning.
00859  *
00860  * Note: channel balancing is enabled by default.
00861  *
00862  * @param bs Handle to the Bitshark device
00863  * @param samplerate Samplerate per second, range: 19,532 to 24,000,000
00864  * @param freq_kHz The frequency in kHz ranging from 300 MHz to 2 GHz
00865  * @param bw_kHz The bandwidth for the low pass filter, the actual bandwidth will be at most samplerate/2 to prevent aliasing
00866  * @param real_samplerate_p Pointer to a unsigned int which will contain the actual sample rate in Hz (as the sample rate will be approximated)
00867  * @param verbose Make the tuning verbose for debugging purposes
00868  * @return 0 if all ok or an error code otherwise

00871     const unsigned adclk_max = BITSHARK_MAX_ADCLK;      // Limited by the FIR filters, not the Bitshark itself
00872     //const int dc_gap = 60000;                         // DC gap which should be avoided (20kHz@-3dB or 60kHz@-0dB)
00873 
00874     // Sanity checking: determine if the sample rate is sane
00875     if(samplerate < 19532 || samplerate > 105000000) {
00876         printf("bitshark_auto_tune2 error: Requested sample rate is out of range [19532, 105000000]: %u S/s\n", samplerate);
00877         return BS_ERR_SAMPLE_RATE;
00878     }
00879     if(freq_kHz < 300000 || freq_kHz > 4000000) {
00880         printf("bitshark_auto_tune2 error: Requested frequency is out of range [300000 kHz, 4000000 kHz]: %u kHz\n", freq_kHz);
00881         return BS_ERR_FREQ;
00882     }
00883     if(bw_kHz < 1 || bw_kHz > (samplerate / 1000) / 2) {
00884         printf("bitshark_auto_tune2 error: Requested bandwidth is out of range [1 kHz, %u kHz @ %u S/s]: %u kHz\n", (samplerate / 1000)/2, samplerate, bw_kHz);
00885         return BS_ERR_BW;
00886     }
00887

00889     // of the spectrum, with an offset to stay away from the DC gap.

00891     int tgt_freq_kHz = freq_kHz - tgt_freq_offset_kHz;                      // The tuner of the Bitshark should be set to this frequency
00892 
00893     // Determine which decimation factor we need to approximate and what the A/D sample frequency will become.
00894     unsigned int dec = (5000000 / samplerate);
00895     // If this is the lower boundary (indicated by a remainder), lower the decimation one step. This will
00896     // allow us to use a higher sample frequency that 5MS/s on the Bitshark ADC.
00897     if(5000000 % samplerate > 0) dec++;
00898 
00899     // With this obtained decimation factor, calculate the required A/D clock frequency in kHz.
00900     unsigned int bs_adclk_kSs = (dec * samplerate) / 1000;
00901 
00902     // Make sure the derrived AD clock is at least twice the requested bandwidth or the tuning will fail
00903     if(bw_kHz*4 > bs_adclk_kSs) {

00911             return BS_ERR_OTHER;

00913     }
00914 
00915     // Calculate what the actual the samplerate is at the outputs with these settings
00916     unsigned int real_samplerate = (bs_adclk_kSs * 1000) / dec;

00918

00920     if(dec >= 256) {
00921         printf("bitshark_auto_tune2 error: required decimation factor %i is out of range\n", dec);
00922         return BS_ERR_OTHER;
00923     }
00924     // Safety check: the FIR filters can only handle rates up to 10 MS/s
00925     if(bs_adclk_kSs > adclk_max/1000) {
00926         printf("bitshark_auto_tune2 error: required A/D clock too high: %i kS/s - max: %i kS/s\n", bs_adclk_kSs, adclk_max/1000);
00927         return BS_ERR_OTHER;
00928     }
00929 
00930     // Set the real sample rate based on the decimation factor
00931     if(real_samplerate_p != NULL) *real_samplerate_p = real_samplerate;
00932 
00933     // Calculate the optimum bandwidth (assuming we want Wn=0.5) for the Bitshark module based on the AD clock
00934     // Note: do not replace bw_kHz as the FIR filter - below - can more precisely handle the low pass stuff.
00935     unsigned bw = bitshark_validate_bandwidth(bs_adclk_kSs);
00936     if(bw > bs_adclk_kSs) {
00937         printf("bitshark_auto_tune2 bug: the generated bandwidth setting violates Niquist!\n");
00938         abort();
00939     }
00940     // End of parameter generation: apply the settings ----------------------------------------------------------------------
00941 
00942     // Tune into the requested frequency
00943     if(verbose) printf("Tuning Bitshark into %i kHz\n", tgt_freq_kHz);

00945 
00946     // Set the A/D and Bitshark filter to the calculated values and set the filter to something sane as well

00993 /**
00994  * Decode a fixed point number into a floating point value. The parameters follow the Xilinx notation scheme for fixed point
00995  * numbers. For example: FIX_10_6. This format is made up of 10 bits, left 4 bits are integer bits (including the sign) and the
00996  * right 6 bits are fractional.
00997  * @param input The unsigned integer holding the bits to decode
00998  * @param bits Number of bits in the fixed point number, this should ofcourse be less than the number of bits 'input' can hold
00999  * @param fractional Number of bits that are fractional of the total number of bits
01000  * @return The floating point value described by the fixed point number
01001  */
01002 float bitshark_decode_fp(unsigned short input, int bits, int fractional) {
01003     int m = bits - fractional;
01004     int n = fractional;
01005 
01006     // Get the first bit
01007     float out = m_get_bit(input, bits-1) * -powf(2, m-1);
01008 
01009     for(int j=-n;j<m-1;j++) {
01010         int i = j+fractional;
01011         //printf("j: %i - i: %i\n", j, i);
01012         out += m_get_bit(input, i) * powf(2, j);
01013     }
01014 
01015     return out;
01016 }

01025 /**
01026  * Load channel balancer with the given values where I is C and Q is D.
01027  * This is a 'fix' for direct conversion receivers as they tend to have small
01028  * channel imbalances which result in reflections between positive and negative frequencies.
01029  * The balancer modifies the Q channel as: Q' = C*I + D*Q.
01030  * The format for C and D is fixed point FIX_10_6 as thats what is returned by the
01031  * Bitshark control device. It will get sign extended before using.
01032  * @param bs Handle to the struct with all control registers
01033  * @param iq The struct to obtain the C and D parameters from. Use NULL to load C=0.0 and D=1.0
01034  * @param verbose Set to true to see what is going on in intermediate steps
01035  */
01036 int bitshark_load_iqbal(bitshark_dev_t *bs, iq_t *iq, int verbose) {
01037     unsigned short C = 0x0, D = 0x40;       // Default: C = 0 and D = 1.0 in FIX_10_6
01038     if(iq) {
01039         C = iq->I;
01040         D = iq->Q;
01041     }
01042     // Sign extend C
01043     {
01044         // Perform sign extending
01045         unsigned short sign = (C & (1 << 9)) != 0 ? 1 : 0;
01046         unsigned short mask = sign == 1 ? 0xfc00 : 0;   // All ones in the unused first bits or all zeros depending on the sign
01047         C = C | mask;
01048     }
01049     // Sign extend D
01050     {
01051         // Perform sign extending
01052         unsigned short sign = (D & (1 << 9)) != 0 ? 1 : 0;
01053         unsigned short mask = sign == 1 ? 0xfc00 : 0;   // All ones in the unused first bits or all zeros depending on the sign
01054         D = D | mask;
01055     }
01056     // Concat both numbers into one 32 bit value
01057     unsigned int CD = (C << 16) | D;
01058 
01059     // Feedback: show what we did
01060     if(verbose) {
01061         float iqC = bitshark_decode_fp(iq->I, 10, 6);
01062         float iqD = bitshark_decode_fp(iq->Q, 10, 6);
01063         printf("[bitshark_set_iqbal] Balance IQ using C=%f (0x%04X) and D=%f (0x%04X)\n", iqC, C, iqD, D);
01064     }
01065 
01066     bs->ctrl->iqbal_reg = CD;
01067     // Readback to test if it worked
01068     if(bitshark_dev.ctrl->iqbal_reg != CD) {
01069         printf("[bitshark_set_iqbal] Read-back error: IQ balance register set to: %08X - should be: %08X\n", (unsigned int)bitshark_dev.ctrl->iqbal_reg, CD);
01070         return BS_ERR_BAL;
01071     }
01072     return BS_NO_ERR;
01073 }
