00001 /*!\file
00002  * \details 2D render library for the XPS DVI interface as exposed in the NEST platform.
00003  *
00004  * This library uses screen subdivisions (or subdivs for short).
00005  * This means that multiple threads can be used to render a subdiv of the screen.
00006  * All functions have a xsubdiv and ysubdiv parameter, which indicate which subdiv is intended to use.
00007  * When set, a thread cannot write outside of its subdiv; all images are clipped.
00008  * When either is set to -1, clipping is overruled.

00871     // line segments to fill.
00872     for (y=IMAGE_TOP; y<IMAGE_BOT; y++) {
00873         //  Build a list of nodes. These are line segments for the given row (y)
00874         nodes = 0;
