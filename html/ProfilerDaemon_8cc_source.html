00036         s[i].tv.tv_usec=0;
00037     }
00038     // prepare for first sample: clear counters and wait
00039     prf_mb_snapshot();
00040     Yield();
00041 
00042     // main poll loop
00043     for(int i=0;i<(int)arg;i++){
00044         usleep(PROFILER_SAMPLE_TIME*1000);
