00574         case LCK_DESTROYED:

00576             // so you will get the lock right now

00578             DEBUGC(lock,'~');

00600             break;
00601         }
00602     }

00604     UNLOCK_LIST(list);

00686 static int lck_pass_p(void* ul,void* l,void* protectee,size_t sprotectee,bool only_try_local){
00687     if(unlikely(!l))
00688         return EINVAL;
00689     LASSERT(getpid()!=0,"Sync daemon itself cannot lock (trying to lock %p)",l);
00690     DEBUG("locking %p...",l);
00691 
00692     int res=0;
00693 
00694     // check initialization
00695     lck_my_locks_t* list=LCK_ADMIN;
00696     if(unlikely(!list->root&&!list->spare)){
00697         // never locked before, initialize now
00698         if((res=dLockRegister(list)))
00699             return res;
00700         if((res=atexit(lck_cleanup)))
00701             return res;
00702         // all fields have been set to 0 by kernel
00703     }
00704 
00705     int lr=0;
00706 
00707     // look for lock in our list
00708     lck_my_lock_t* lock=lck_list_find(list,l);
00709     if(likely(lock!=NULL)){
00710         // we have one in our list, try locking
00711         switch((lck_lock_state_t)lcas(&lock->state_,LCK_FREE,LCK_LOCKED)){
00712         case LCK_FREE:
00713             // we locked it
00714             DEBUG("Lock %p: locked",l);
00715             // locking a local lock within a signal handler is safe (but should still be avoided)
00716             lck_list_use(list,lock);
00717             mc_reprotect(lock,protectee,sprotectee,-1);
00718             if(ul&&(res=lck_unlock(ul))>0)
00719                 return res;
00720             return -LR_LOCAL;
00721         case LCK_LOCKED:
00722             // recursive locking is not allowed
00723             DEBUG("Lock %p: recursive",l);
00724             return EDEADLK;
00725         case LCK_MIGRATED:
00726             // since we must migrate the lock, this should not be possible
00727             LASSERT(false,"Locking migrated lock is not possible");
00728             return EINVAL;
00729         case LCK_STOLEN:
00730             if(only_try_local)
00731                 return EREMOTE;
00732             CHECK_NON_SIGHANDLER;
00733             // lock has been stolen, we don't have the lock anymore, reuse this record
00734             DEBUG("Lock %p: stolen (assume remote)",l);
00735             lck_list_use(list,lock);
00736             lock->own=LCK_MINE;
00737             lock->state=LCK_LOCKED;
00738             lr|=LR_STOLEN;
00739             break;
00740         case LCK_DESTROYED:
00741             // not possible
00742             LASSERT(false,"Cannot lock destroyed lock %p@%p",l,lock);
00743             return EINVAL;
00744         }
00745     }else{
00746         if(only_try_local)
00747             return EREMOTE;
00748         CHECK_NON_SIGHANDLER;
00749         DEBUG("Lock %p: remote",l);
00750         // cannot find lock, create new record
00751         if(!(lock=lck_list_add(list,l)))
00752             return ENOMEM;
00753         // lck_list_add has set the default values just before adding it to the list:
00754         // lock->own=LCK_MINE;
00755         // lock->state=LCK_LOCKED;
00756         lr|=LR_UNKNOWN;
00757     }
00758 
00759     LASSERT(!only_try_local,"Fetching local lock from remote...");
00760 
00761     // We don't have the lock, request it from server.
00762     // lock points to a new record that can be used for that.
00763     LASSERT(lock->own==LCK_MINE&&lock->state==LCK_LOCKED,"Invalid state of new record: %d %d",lock->own,lock->state);
00764 
00765     if((res=dLockRequest(l))==0){
00766         // got lock
00767         DEBUG("Lock %p: got it, did not exist yet",l);
00768         mc_protect(lock,protectee,sprotectee);
00769         if(ul&&(res=lck_unlock(ul))>0)
00770             return res;
00771         return -(lr|LR_NEW);
00772     }else if(res>0){
00773         // error
00774         DEBUG("Lock %p: request error %d",l,res);
00775         return res;
00776     }else{
00777         // go ask somewhere else
00778         LASSERT(-res!=getgpid(),"Cannot ask myself for lock");
00779         DEBUG("Asking %d for lock %p",-res,l);
00780         switch((res=dLockAsk(l,-res))){
00781         case 0:
00782             // lock stolen
00783             DEBUG("Lock %p: I stole lock",l);
00784             mc_protect(lock,protectee,sprotectee);
00785             if(ul&&(res=lck_unlock(ul))>0)
00786                 return res;
00787             return -(lr|LR_FREE);
00788         case EBUSY:{
00789             // lock locked, wait for signal
00790             if(ul&&(res=lck_unlock(ul))>0)
00791                 return res;
00792             DEBUG("Lock %p: waiting...",l);
00793             sigset_t s;
00794             sigemptyset(&s);
00795             sigaddset(&s,SIGLCK);
00796             // when woken up, we got the lock
00797             if((res=Suspend(NULL,&s))!=EINTR)
00798                 return res;
00799             mc_protect(lock,protectee,sprotectee);
00800             return -(lr|LR_LOCKED);
00801             }
00802         default:
00803             // error

00805             return res;
00806         }
00807     }

00822  * Does not communicate with lock daemon, unless old lock is destroyed.

00898 /*! \brief Unlock and give back to server
00899  * When result < 0:
00900  * LR_LOCAL: lock was local
00901  * LR_STOLEN: already stolen before destruction (no communication with lock daemon)
00902  * LR_STEALING: theft in progress, yielding until stolen (after giveup)
00903  * LR_MIGRATED: migrated (after giveup)
00904  * possible combinations: LR_STOLEN || (LR_LOCAL|(LR_STEALING?||LR_MIGRATED))
00905  */
00906 int lck_destroy(void* l){
00907     LASSERT(getpid()!=0,"Sync daemon itself cannot destroy locks (trying to destroy %p)",l);
00908     CHECK_NON_SIGHANDLER;
00909 
00910     lck_my_locks_t* list=LCK_ADMIN;
00911     lck_my_lock_t* lock=lck_list_find(list,l);
00912 
00913     if(!lock)
00914         return ESRCH;
00915     
00916     int res=0;
00917     int lr=0;
00918 
00919     // remove from list (locks list, so atomically w.r.t. lck_ask)
00920     lck_list_remove(list,lock);
00921 
00922     // get lock
00923     switch(lock->state){

00925         // lock was free, giveup

00929     case LCK_LOCKED:
00930         // lock was locked, check for waiting process
00931         lr|=LR_LOCAL;
00932         // migrate when needed

00934         case LCK_MINE:
00935             // nobody was interested, giveup
00936             goto giveup;
00937         case LCK_REQUEST:
00938             // migrate immediately
00939             DEBUG("Destroy %p: migrate",l);
00940             // wakeup waiting process
00941             res=lck_migrate(lock);
00942             lr|=LR_MIGRATED;
00943             // giveup not needed after migration, we are done
00944             goto done;
00945         }
00946         // unreachable
00947         goto done;
00948     case LCK_STOLEN:
00949         // just got stolen, we are done
00950         DEBUG("Destroy %p: stolen",l);
00951         lck_list_use(list,lock);
00952         DEBUGC(lock,'\\');
00953         lr|=LR_STOLEN;
00954         goto done;
00955     case LCK_MIGRATED:
