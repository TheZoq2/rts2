00534 /******************************************************************************
00535  * compact_node - Tries to merge the previous and current records with the
00536  *                specified one. Assumes that the ptr is correctly in the
00537  *                address sorted free list.
00538  */
00539 static void compact_node(HEAP* const heap,HEAP_NODE* ptr){
00540     HEAP_NODE *pPrev, *pNext;
00541     pPrev = ptr->pPrevAddr;
00542     pNext = ptr->pNextAddr;
00543 
00544     /* First to the node before us and calc addr at end of block and see if it's this one. */
00545     if(pPrev != NULL && ptr == (HEAP_NODE*)((char*)pPrev + pPrev->nLength + HEAP_NODE_SIZE)){
00546         /* Found contiguous nodes - merge them */
00547         pPrev->nLength += (HEAP_NODE_SIZE + ptr->nLength);
00548         pPrev->pNextAddr = ptr->pNextAddr;
00549         if (ptr->pNextAddr != NULL)
00550             ptr->pNextAddr->pPrevAddr = ptr->pPrevAddr;
00551 
00552         remove_from_size_list(heap,ptr);
00553         ptr = pPrev;
00554     }
00555 
00556     if (pNext != NULL && pNext == (HEAP_NODE*)((char*)ptr + ptr->nLength + HEAP_NODE_SIZE)){
00557         /* Found contiguous nodes - merge them */
00558         ptr->nLength += (HEAP_NODE_SIZE + pNext->nLength);

00561             ptr->pNextAddr->pPrevAddr = ptr;

00564     }
