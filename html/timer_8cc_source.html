00016  * \brief Sets the local timer to the designated number of clock cycles.
00017  * \details The timer will be started automatically. When the value 0 is
00018  *          reached, an interrupt is generated.
00019  * \param time number of cycles to set the counter to
00020  * \param repeat set to non-zero to reload and restart the timer with this interval when zero is reached
00021  */
00022 void kSetTimer(unsigned int time,unsigned int repeat){
00023 
00024     // Set MSR[BIP] to disable interrupts. MSR[IE] cannot be used, because
00025     // kSetTimer is called from within the interrupt handler.
00026     // Setting MSR[BIP] ensures that interrupts, breaks, etc. are disabled for all
00027     // subsequent instructions, hence dummy nops are not neccessary.
00028     // As breaks are not used by the kernel anywhere else, it is safe to use it here.
00029     asm volatile ("msrset r0, 1<<3");
00030 
00031     const unsigned int flags=
00032         (0<<7)|     // disable timer
00033         (1<<6)|     // enable interrupt
00034         (0<<5)|     // do not load timer from TLR
00035         (0<<4)|     // do not reload timer
00036         (0<<3)|     // disable external capture
00037         (1<<2)|     // enable external generate signal
00038         (1<<1)|     // down counter
00039         (0<<0);     // generate mode
00040 
00041     // stop counter
00042     io_write(IO_TIMER_TCSR0,unsigned int,flags|
00043         (1<<8)|     // clear interrupt flag
00044         (1<<5));    // load timer from TLR
00045     // load value
00046     io_write(IO_TIMER_TLR0,unsigned int,time);
00047 
00048     if(repeat){
00049         // do not load timer from TLR anymore
00050         io_write(IO_TIMER_TCSR0,unsigned int,flags|
00051             (0<<5));    
00052         // load interval value
00053         io_write(IO_TIMER_TLR0,unsigned int,repeat);
00054     }
00055 
00056     // start counter
00057     io_write(IO_TIMER_TCSR0,unsigned int,flags|
00058         (1<<7)|     // enable timer
00059         ((repeat?1:0)<<4)); // repeat when neccessary
00060 
00061     // Unset MSR[BIP] to enable interrupts again.
00062     asm volatile ("msrclr r0, 1<<3");
00063 }
